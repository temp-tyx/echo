### 2.2 _apply_echo_pruning (核心剪枝逻辑)
```python
def _apply_echo_pruning(self, draft_token_ids: torch.Tensor, batch_size: int) -> torch.Tensor:
    if not hasattr(self, "_echo_logits_list") or not self._echo_logits_list:
        return draft_token_ids

    # 1. 计算每步的条件log-prob
    logits_list = self._echo_logits_list
    log_probs_per_step = []
    for step_idx in range(total_steps):
        logits = logits_list[step_idx]
        log_probs = F.log_softmax(logits, dim=-1)
        step_tokens = draft_token_ids[:, step_idx]
        step_log_probs = log_probs.gather(1, step_tokens.unsqueeze(1)).squeeze(1)
        log_probs_per_step.append(step_log_probs)

    # 2. cumsum得累积概率
    cond_log_probs = torch.stack(log_probs_per_step, dim=1)
    cum_log_probs = torch.cumsum(cond_log_probs, dim=1)

    # 3. global top-k (考虑echo_keep约束)
    k_max = envs.VLLM_ECHO_K_MAX
    flat_log_probs = cum_log_probs.flatten()
    _, top_flat_indices = torch.topk(flat_log_probs, k=k_max)

    # 4. mask并返回
    mask = torch.zeros_like(flat_log_probs, dtype=torch.bool)
    mask[top_flat_indices] = True
    mask = mask.view(actual_batch_size, total_steps)

    pruned_draft_ids = draft_token_ids.clone()
    pruned_draft_ids[~mask] = -1
    return pruned_draft_ids
```

### 3.1 execute_model开头trim (已修改)
```python
if (
    envs.VLLM_ECHO_ENABLED
    and self.echo_cu_draft_tokens
    and len(scheduler_output.num_scheduled_tokens) != 0

    for req_id in list(scheduler_output.num_scheduled_tokens.keys()):
        draft_token_num = self.echo_cu_draft_tokens.get(req_id)
        if draft_token_num is None:
            continue
        # 修剪scheduled_spec_decode_tokens
        spec_tokens = scheduler_output.scheduled_spec_decode_tokens.get(req_id, [])
        scheduler_output.scheduled_spec_decode_tokens[req_id] = spec_tokens[:draft_token_num]
        # 更新num_scheduled_tokens
        old_num = scheduler_output.num_scheduled_tokens[req_id]
        scheduler_output.num_scheduled_tokens[req_id] = draft_token_num + 1
        scheduler_output.total_num_scheduled_tokens -= (old_num - (draft_token_num + 1))
    self.echo_cu_draft_tokens = {}
```

### 3.2 调试日志 (已添加)

三个关键日志点：

1. **`ECHO [slot_mapping]`** - 在`_prepare_inputs`后
   - 检查position、slot、block_table一致性

2. **`ECHO [prepare]`** - 在`_prepare_inputs`末尾
   - 检查attn_state、seq_lens_gpu/cpu、seq_drift

3. **`ECHO [attn]`** - 在`_build_attention_metadata`后
   - 检查attention实际使用的metadata

4. **`ECHO [forward]`** - 在`_model_forward`后
   - 检查hidden_states的NaN情况
